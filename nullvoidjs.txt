    function shuffle(a) {
      var j, x, i;
      for (i = a.length - 1; i > 0; i--) {
        j = Math.floor(Math.random() * (i + 1));
        x = a[i];
        a[i] = a[j];
        a[j] = x;
      }
      return a;
    }

    /* this is a stand in function for trevors distance code that we need to use to compare the words together */
    function correctLetterCount(firstWord, secondWord) {
      if (firstWord.length !== secondWord.length) {
        return 0;
      }
      var correctCount = 0;
      var i = 0;
      while (i < firstWord.length) {
        if (firstWord[i] == secondWord[i])
          correctCount++;
        i++;
      }
      return correctCount;

    }
